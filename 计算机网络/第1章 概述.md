<template>
	<view class="login-container">
		<view class="input-item">
			<input v-model="username" placeholder="请输入用户名" />
		</view>
		<view class="input-item">
			<input v-model="password" type="password" placeholder="请输入密码" />
		</view>
		<button @click="handleLogin" class="login-btn">登录</button>
		<view class="error-tip" v-if="errorMsg">{{ errorMsg }}</view>
	</view>
</template>

<script>
	// 直接使用原生JavaScript的XMLHttpRequest，彻底绕过UniApp的请求封装
	export default {
		data() {
			return {
				username: '',
				password: '',
				errorMsg: ''
			};
		},
		methods: {
			handleLogin() {
				this.errorMsg = '';
				if (!this.username || !this.password) {
					this.errorMsg = '用户名/密码不能为空';
					return;
				}

				// 1. 构建x-www-form-urlencoded参数（与Postman完全一致）
				const params = new FormData();
				params.append('username', this.username);
				params.append('password', this.password);

				// 2. 原生XMLHttpRequest，无任何框架封装
				const xhr = new XMLHttpRequest();
				xhr.open('POST', 'http://localhost:8080/auth/login', true);
				// 3. 设置请求头（与Postman一致）
				xhr.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded');
				xhr.onreadystatechange = () => {
					if (xhr.readyState === 4) {
						if (xhr.status === 200) {
							// 登录成功，解析Token（与Postman响应格式一致）
							const res = JSON.parse(xhr.responseText);
							// 保存Token到本地（替代Pinia，避免状态管理干扰）
							uni.setStorageSync('token', res.token);
							uni.showToast({
								title: '登录成功',
								icon: 'success'
							});
							uni.switchTab({
								url: '/pages/student/student'
							});
						} else {
							// 解析后端返回的错误信息
							const errRes = JSON.parse(xhr.responseText);
							this.errorMsg = errRes.msg || `请求失败，状态码：${xhr.status}`;
						}
					}
				};
				// 4. 发送请求（与Postman参数一致）
				xhr.send(new URLSearchParams(params).toString());
			}
		},
		onShow() {
			// 清除本地无效Token
			uni.removeStorageSync('token');
		}
	};
</script>
